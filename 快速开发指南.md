### 选项4: 测试WebSocket功能

```typescript
// 在聊天页面使用WebSocket
import { useWebSocket } from '@/hooks/useWebSocket'

const ChatPage = () => {
  const { isConnected, on, emit } = useWebSocket()

  // 监听消息
  on('new_message', (data) => {
    console.log('收到消息:', data)
  })

  // 发送消息
  const sendMessage = () => {
    emit('send_message', {
      to: 'user123',
      content: '你好!'
    })
  }

  return (
    <View>
      <Text>状态: {isConnected() ? '已连接' : '未连接'}</Text>
      <Button onClick={sendMessage}>发送消息</Button>
    </View>
  )
}
```
